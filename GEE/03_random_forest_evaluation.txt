function trainRandomForest(
  features,
  seedValue
) {

  return ee.Classifier.smileRandomForest({
    numberOfTrees 100,
    variablesPerSplit 3,
    minLeafPopulation 5,
    bagFraction 0.5,
    maxNodes 20,
    seed seedValue
  }).train({
    features features,
    classProperty 'class',
    inputProperties featureBands
  });
}


var rfModel = trainRandomForest(
  trainingData,
  seed
);

var learningTrainingData =
  trainingData.randomColumn(
    'learning_rand',
    seed + 100
  );


var learningFractions = ee.List([
  0.25,
  0.50,
  0.75,
  1.00
]);


var learningCurve = ee.FeatureCollection(
  learningFractions.map(
    function(fraction) {

      fraction =
        ee.Number(fraction);


      var trainingSubset =
        learningTrainingData.filter(
          ee.Filter.lt(
            'learning_rand',
            fraction
          )
        );


      var learningModel =
        trainRandomForest(
          trainingSubset,
          seed
        );


      var learningTrainingMatrix =
        trainingSubset
          .classify(
            learningModel
          )
          .errorMatrix(
            'class',
            'classification',
            [0, 1]
          );


      var learningTestingMatrix =
        testingData
          .classify(
            learningModel
          )
          .errorMatrix(
            'class',
            'classification',
            [0, 1]
          );


      return ee.Feature(
        null,
        {

          training_fraction
            fraction,

          training_percent
            fraction.multiply(100),

          training_samples
            trainingSubset.size(),

          training_accuracy
            learningTrainingMatrix
              .accuracy(),

          validation_accuracy
            learningTestingMatrix
              .accuracy()
        }
      );
    }
  )
);

var overallAccuracy =
  errorMatrix.accuracy();


var kappaIndex =
  errorMatrix.kappa();


var orderList =
  errorMatrix.order();


var orderKeys =
  orderList.map(
    function(v) {

      return ee.Number(v)
        .format();
    }
  );


var consumersAccuracyArray =
  errorMatrix
    .consumersAccuracy();


var producersAccuracyArray =
  errorMatrix
    .producersAccuracy();


 Precision
 array 1 x n, ambil baris pertama
var precisionList = ee.List(
  consumersAccuracyArray
    .toList()
    .get(0)
);


 Recall
 array n x 1, setiap baris satu elemen
var recallRows =
  producersAccuracyArray
    .toList();


var recallList =
  recallRows.map(
    function(row) {

      return ee.List(row)
        .get(0);
    }
  );


var precisionDict =
  ee.Dictionary.fromLists(
    orderKeys,
    precisionList
  );


var recallDict =
  ee.Dictionary.fromLists(
    orderKeys,
    recallList
  );


 Ambil nilai kelas Vegetasi
 label 1
var precisionVegetasi =
  ee.Number(
    precisionDict.get(
      '1',
      -1
    )
  );


var recallVegetasi =
  ee.Number(
    recallDict.get(
      '1',
      -1
    )
  );


var f1ScoreVegetasi =
  ee.Algorithms.If(

    precisionVegetasi
      .eq(-1)
      .or(
        recallVegetasi.eq(-1)
      ),

    -1,

    ee.Number(2)
      .multiply(
        precisionVegetasi
          .multiply(
            recallVegetasi
          )
      )
      .divide(
        precisionVegetasi
          .add(
            recallVegetasi
          )
      )
  );


f1ScoreVegetasi =
  ee.Number(
    f1ScoreVegetasi
  );


print(
  '=== HASIL EVALUASI MODEL (Kelas Vegetasi = label 1) ==='
);


print(
  'Jumlah kelas terdeteksi',
  orderList.size()
);


print(

  overallAccuracy
);


print(
  '2. Precision (Vegetasi)',

);

  '3. Recall (Vegetasi)',

var matrixArray =
  errorMatrix.array();


var matrixList =
  matrixArray.toList();


var rowDict =
  ee.Dictionary.fromLists(
    orderKeys,
    matrixList
  );


function getCell(
  actualLabel,
  predictedLabel
) {

  var rowRaw =
    rowDict.get(
      actualLabel,
      null
    );


  var row =
    ee.Algorithms.If(
      rowRaw,
      ee.List(rowRaw),
      null
    );


  var colDict =
    ee.Algorithms.If(

      row,

      ee.Dictionary.fromLists(
        orderKeys,
        ee.List(row)
      ),

      ee.Dictionary({})
    );


  return ee.Number(
    ee.Dictionary(
      colDict
    ).get(
      predictedLabel,
      -1
    )
  );
}


var truePositive =
  getCell(
    '1',
    '1'
  );


var trueNegative =
  getCell(
    '0',
    '0'
  );


var falsePositive =
  getCell(
    '0',
    '1'
  );


var falseNegative =
  getCell(
    '1',

 11. PENILAIAN KELAYAKAN MODEL
 =====================================================

overallAccuracy.evaluate(
  function(oa) {

    kappaIndex.evaluate(
      function(kappa) {

        orderList.size().evaluate(
          function(n) {

            print(
              '=== PENILAIAN KELAYAKAN MODEL ==='
            );


            print(
              'Overall Accuracy ' +
              (oa  100)
                .toFixed(2) +
              '%'
            );


            print(
              'Kappa Index ' +
              kappa.toFixed(3)
            );


            if (n  2) {

              print(
                '❌ Model TIDAK DAPAT dinilai kelayakannya secara valid untuk 2 kelas.'
              );

              return;
            }


            var kelayakan;


            if (
              oa = 0.85 &&
              kappa = 0.61
            ) {

              kelayakan =
                'Model CUKUP LAYAK dipakai untuk analisis perubahan tutupan ' +
                'vegetasi 2024-2025. Akurasi keseluruhan tinggi dan nilai Kappa menunjukkan ' +
                'kesepakatan yang kuat antara hasil klasifikasi dan data referensi.';

            } else if (
              oa = 0.70 &&
              kappa = 0.41
            ) {

              kelayakan =
                'Model CUKUP DAPAT DIGUNAKAN namun dengan CATATAN. Akurasi dan ' +
                'Kappa berada pada kategori sedang, hasil klasifikasi memberikan gambaran ' +
                'umum pola perubahan namun perlu verifikasi lebih lanjut.';

            } else {

              kelayakan =
                'Model BELUM CUKUP LAYAK dipakai untuk analisis perubahan secara ' +
                'kuantitatif. Disarankan meninjau ulang kualitas dan jumlah sampel trainingtesting.';
            }


            print(
              kelayakan
            );
          }
        );
      }
    );
  }
);
